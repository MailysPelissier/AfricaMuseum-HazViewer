CREATE MATERIALIZED VIEW IF NOT EXISTS vue_paragraphs_pg AS
SELECT 
    (p.paragraph_id || '_' || ep.event_id) AS composite_id, ep.event_id, p.*
FROM 
    paragraphs p
JOIN 
    event_paragraphs ep ON p.paragraph_id = ep.paragraph_id;
